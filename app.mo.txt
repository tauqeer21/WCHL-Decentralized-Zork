import Text "mo:base/Text";
import Nat "mo:base/Nat";

actor {
  // --- Game State ---
  var location : Nat = 0;
  var hasKeycard : Bool = false;

  // --- Main Game Loop ---
  public func processCommand(rawInput: Text) : async Text {
    let input = Text.toLowercase(rawInput);

    // --- Command: "look" ---
    if (input == "look") {
      if (location == 0) {
        var response = "You are in a cold, loud Server Room. Blinking lights reflect off the polished floor. There is a door to the north.";
        if (not hasKeycard) {
          response #= " You see a forgotten keycard on a server rack.";
        };
        return response;
      } else if (location == 1) {
        return "You are in front of a massive, humming Mainframe. The air crackles with energy. A door leads south.";
      } else {
        return "You are in a cramped Ventilation Shaft. It is dark and dusty. A faint light comes from the west.";
      };
    };

    // --- Command: "north" ---
    if (input == "north") {
      if (location == 0) {
        if (hasKeycard) {
          location := 1; // Move to Mainframe
          return "The door slides open with a hiss. You enter the Mainframe room.";
        } else {
          return "The door is locked. A keycard is required.";
        };
      } else {
        return "You can't go north from here.";
      };
    };

    // --- Command: "south" ---
    if (input == "south") {
      if (location == 1) {
        location := 0; // Move to Server Room
        return "You return to the Server Room.";
      } else {
        return "You can't go south from here.";
      };
    };

    // --- Command: "get keycard" ---
    if (input == "get keycard") {
      if (location == 0 and not hasKeycard) {
        hasKeycard := true;
        return "You picked up the keycard.";
      } else if (hasKeycard) {
        return "You already have the keycard.";
      } else {
        return "You don't see a keycard here.";
      };
    };

    // --- Unknown Command ---
    return "Unknown command. Try 'look', 'north', 'south', or 'get keycard'.";
  };
}
